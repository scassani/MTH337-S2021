[39]:

def divisors(n):
    '''This function returns all divisors of n
    it uses a for loop''' # docstring
    mylist=[] # define empty list to store divisors
    for k in range(1,n+1): # loop through possible divisors
        if n%k==0: # check if k divides n
            mylist.append(k)
    return mylist
