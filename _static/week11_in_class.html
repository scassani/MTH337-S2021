def logplot(x,y,t=""):
    

    plt.figure(figsize=(12,6))
    plt.subplot(141)
    plt.plot(x,y)
    plt.title("x ; y")
    plt.subplot(142)
    plt.plot(np.log(x),y)
    plt.title("log-x ; y")
    plt.subplot(143)
    plt.plot(x,np.log(y))
    plt.title("x ; log-y")
    plt.subplot(144)
    plt.plot(np.log(x),np.log(y))
    plt.title("log-x ; log-y")
    
    plt.suptitle(t)

    return None
