%matplotlib inline
%config InlineBackend.figure_format = 'retina'
import matplotlib.pyplot as plt
import numpy as np


v = [10, 11, 12, 13, 14, 15]
g = 9.81

plt.figure(figsize=(8, 2.7))
plt.style.use('classic')
plt.xlim(0, 3.5)
plt.ylim(0, 12)
t = np.linspace(0, 3.5, 100)
for v_0 in v :
    plt.plot(t, v_0*t - 0.5*g*t**2, 'b-', lw=1)
plt.show()
